\section{Umsetzung}
\subsection{Projektrahmen}


\begin{frame}{Geplanter Rahmen des Projektes}
	\begin{itemize}
		\item Erstellung einer Benutzeroberfläche zur XML generierung für niagarino
		\item XML's sollen Format-technisch direkt in niagarino Ausführbar sein (Vorlage war XML-Datei)
		\item Benutzeroberfläche basierend auf Framework für Graphen und Swing
		\item Wichtigste Features: Zooming, Panning, einfügen von Knoten in anderen Knoten
	\end{itemize}
\end{frame}

\subsection{Aufgetretene Schwierigkeiten}
\begin{frame}{Aufgetretene Schwierigkeiten}
	\begin{itemize}
		\item Eigener Mangel an praktischer Erfahrung
		\item Viele Frameworks: JGraphT, JGraphX, JUNG, JKanvas
		\begin{itemize}
			\item[$\rightarrow$] Einige kleine Prototypen erstellt, um Funktionalitäten zu testen
		\end{itemize}
		\item Keines unterstützt alle Requirements
		\begin{itemize}
			\item[$\rightarrow$] Min. drei Monate verschwendet mit Suche nach dem richtigen Framework
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Aufgetretene Schwierigkeiten}
	\begin{itemize}
		\item Aufgrund von Panning, Zooming und mangels guter Alternativen bei JKanvas geblieben
		\item Kaum Dokumentiert da privates Projekt von Josua Krause
		\item Viele Möglichkeiten, aber wichtige Teile fehlen oder anderst implementiert
		\begin{itemize}
			\item[$\rightarrow$] Fast alle Funktionalitäten mussten angepasst oder selbst implementiert werden
		\end{itemize}
		\item Nicht alle Operatoren im XML definiert und wie in niagarino umgesetzt
	\end{itemize}
\end{frame}


\begin{frame}{Einschränkung des Projektrahmens}
	\begin{itemize}
		\item Umfang wurde zu groß im Rahmen eines Bachelorprojekts
		\item Viele Einschränkungen zu beachten
	\end{itemize}
\end{frame}
\subsection{Proof of Concept}
\begin{frame}{Proof of Concept?}
	\begin{itemize}
		\item Erstellen eines Proof of Concepts das veranschaulicht, dass und wie eine dynamische Lösung möglich ist
		\item Nutzen von Java Annotationen, um herauszufinden wie ein Operator instanziiert werden kann
		\begin{itemize}
			\item[$\rightarrow$] Später könnte niagarino so von NiaGUI angesteuert werden
			% Erklären wie es funktioniert
		\end{itemize}
		\item Proof of Concept anhand von 2 Operatoren (Selection, TopK) mit GUI und Workflow
		\item Was muss noch getan werden, um weitere Operatoren in GUI zu integrieren
	\end{itemize}
\end{frame}

\begin{frame}{Proof of Concept!}
	\begin{itemize}
		\item Die zwei Operatoren wurden ausgewählt, um Machbarkeit, Prinzip und Weg zu veranschaulichen
		\item Nutzen von Reflections, um auf Operatoren zuzugreifen und sie \textit{auszulesen} und nutzbar zu machen
		\begin{itemize}
			\item[$\rightarrow$] Ändern sich die Operatoren in niagarino ändert sich die GUI automatisch und angepasstes XML wird generiert
			% Erklären wie es funktioniert
